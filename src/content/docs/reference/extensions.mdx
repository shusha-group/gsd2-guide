import extensions from '../../../../content/generated/extensions.json';
import ReferenceCard from '../../../../src/components/ReferenceCard.astro';
import ReferenceGrid from '../../../../src/components/ReferenceGrid.astro';
import ToolList from '../../../../src/components/ToolList.astro';

export const sorted = [...extensions].sort((a, b) => b.tools.length - a.tools.length);
export const categories = ["Has Tools", "No Tools"];

Extensions are built-in capabilities bundled with the pi agent. Each extension registers one or more tools the agent can call, or provides hooks, commands, and stream-level guardrails that operate without direct tool calls.

20 extensions are active by default. Cards are sorted by tool count — extensions with more tools appear first. Click any card to expand and see its registered tools.

<ReferenceGrid categories={categories}>
  {sorted.map((ext) => (
    <ReferenceCard
      title={ext.name}
      subtitle={ext.description || "No description available"}
      badge={ext.tools.length > 0 ? ext.tools.length + (ext.tools.length === 1 ? " tool" : " tools") : undefined}
      category={ext.tools.length > 0 ? "Has Tools" : "No Tools"}
    >
      {ext.tools.length > 0 ? (
        <ToolList tools={ext.tools} />
      ) : (
        <p><em>This extension uses hooks or commands instead of registered tools.</em></p>
      )}
    </ReferenceCard>
  ))}
</ReferenceGrid>
