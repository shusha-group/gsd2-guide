import skills from '../../../../content/generated/skills.json';
import ReferenceCard from '../../../../src/components/ReferenceCard.astro';

export const topLevel = skills.filter(s => !s.parentSkill);
export const getSubSkills = (name) => skills.filter(s => s.parentSkill === name);

8 bundled skills that extend the agent's capabilities. Skills are auto-detected based on task context and activated when needed. Click any card to expand for details.

{topLevel.map((skill) => (
  <>
    <ReferenceCard title={skill.name} subtitle={skill.description}>
      {skill.objective && (
        <div>
          <strong>Objective</strong>
          <div style="white-space: pre-wrap; margin-top: 0.25rem;">{skill.objective}</div>
        </div>
      )}
      {skill.arguments && (
        <div style="margin-top: 0.75rem;">
          <strong>Arguments</strong>
          <div style="white-space: pre-wrap; margin-top: 0.25rem;">{skill.arguments}</div>
        </div>
      )}
      {skill.detection && (
        <div style="margin-top: 0.75rem;">
          <strong>Detection</strong>
          <div style="white-space: pre-wrap; margin-top: 0.25rem;">{skill.detection}</div>
        </div>
      )}
    </ReferenceCard>
    {getSubSkills(skill.name).map((sub) => (
      <div class="sub-skill-indent">
        <ReferenceCard title={sub.name} subtitle={sub.description} badge={"Sub-skill of " + skill.name}>
          {sub.objective && (
            <div>
              <strong>Objective</strong>
              <div style="white-space: pre-wrap; margin-top: 0.25rem;">{sub.objective}</div>
            </div>
          )}
          {sub.arguments && (
            <div style="margin-top: 0.75rem;">
              <strong>Arguments</strong>
              <div style="white-space: pre-wrap; margin-top: 0.25rem;">{sub.arguments}</div>
            </div>
          )}
          {sub.detection && (
            <div style="margin-top: 0.75rem;">
              <strong>Detection</strong>
              <div style="white-space: pre-wrap; margin-top: 0.25rem;">{sub.detection}</div>
            </div>
          )}
        </ReferenceCard>
      </div>
    ))}
  </>
))}

<style>{`
  .sub-skill-indent {
    margin-left: 1.5rem;
    border-left: 2px solid var(--terminal-green, #00ff41);
